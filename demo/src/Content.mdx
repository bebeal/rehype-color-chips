<h1>Rehype Color Chips</h1>
<table>
  <tr>
    <td>
      <code>\:#f00</code> → #f00<br />
      <code>\:#0f0</code> → #0f0<br />
      <code>\:#00f</code> → #00f
    </td>
    <td>
      <code>\\::#f00</code> → \:#f00<br />
      <code>\\::#0f0</code> → \:#0f0<br />
      <code>\\::#00f</code> → \:#00f
    </td>
  </tr>
  <tr>
    <td>
      <code>\:rgb(255, 0, 0)</code> → rgb(255, 0, 0)<br />
      <code>\:rgb(0, 255, 0)</code> → rgb(0, 255, 0)<br />
      <code>\:rgb(0, 0, 255)</code> → rgb(0, 0, 255)
    </td>
    <td>
      <code>\\::rgb(255, 0, 0)</code> → \:rgb(255, 0, 0)<br />
      <code>\\::rgb(0, 255, 0)</code> → \:rgb(0, 255, 0)<br />
      <code>\\::rgb(0, 0, 255)</code> → \:rgb(0, 0, 255)
    </td>
  </tr>
</table>
